    8일차: 템플릿 데이터 연결하여 만들어보기

    핵심개념

    정적 페이지와 동적 데이터의 결합

    웹 페이지는 고정된 HTML 구조(템플릿)와 변경되는 데이터를 분리하여 관리합니다. 이를 통해 같은 디자인으로 다양한 데이터를 표시할 수 있습니다.

    정적 페이지 vs 동적 페이지:

    구분 정적 페이지 동적 페이지
    콘텐츠 고정된 내용 데이터에 따라 변경
    수정 방법 HTML 직접 수정 데이터만 수정
    예시 회사 소개 페이지 날씨 대시보드, 뉴스 피드
    장점 단순함, 빠름 유연함, 재사용 가능

    템플릿 + 데이터 결합 구조:

    ┌─────────────────┐      ┌─────────────────┐
│    템플릿        │      │     데이터       │
│   (HTML/CSS)    │  +   │    (JSON)       │
│                 │      │                 │
│  ┌───────────┐  │      │ {               │
│  │ {제목}    │  │      │   "제목": "...",│
│  │ {내용}    │  │      │   "내용": "..." │
│  └───────────┘  │      │ }               │
└─────────────────┘      └─────────────────┘
         │                       │
         └───────────┬───────────┘
                     ▼
         ┌─────────────────────┐
         │    완성된 웹 페이지   │
         │  ┌───────────────┐  │
         │  │ 실제 제목      │  │
         │  │ 실제 내용      │  │
         │  └───────────────┘  │
         └─────────────────────┘

    왜 분리하는가: - 디자이너는 HTML/CSS만, 개발자는 데이터만 관리 - 데이터 변경 시 코드 수정 불필요 - 같은 템플릿으로 여러 페이지 생성 가능 - API에서 받은 데이터를 바로 표시 가능

    데이터 바인딩 기초

    데이터 바인딩(Data Binding)은 데이터와 UI 요소를 연결하는 것입니다. 데이터가 변경되면 화면도 자동으로 업데이트됩니다.

    JavaScript로 데이터 바인딩하기:

    // 1. 데이터 준비
const data = {
    title: "오늘의 날씨",
    temperature: 15.5,
    description: "맑음"
};

// 2. HTML 요소 선택
const titleElement = document.getElementById("title");
const tempElement = document.getElementById("temperature");

// 3. 데이터 바인딩 (연결)
titleElement.textContent = data.title;
tempElement.textContent = data.temperature + "°C";

    데이터 바인딩 방식:

    방식 설명 예시
    innerHTML HTML 태그 포함 삽입 el.innerHTML = "<b>굵게</b>"
    textContent 텍스트만 삽입 (안전) el.textContent = "텍스트"
    setAttribute 속성 변경 img.setAttribute("src", url)
    style 스타일 변경 el.style.color = "red"
    classList 클래스 추가/제거 el.classList.add("active")

    JSON 데이터 로드 방법:

    // 방법 1: fetch API (권장)
fetch('data.json')
    .then(response => response.json())
    .then(data => {
        console.log(data);
        // 데이터 바인딩 수행
    });

// 방법 2: async/await (더 깔끔)
async function loadData() {
    const response = await fetch('data.json');
    const data = await response.json();
    return data;
}

// 방법 3: 변수로 직접 선언 (테스트용)
const data = {
    "name": "홍길동",
    "age": 25
};

    배열 데이터 반복 렌더링:

    const items = [
    { name: "사과", price: 1000 },
    { name: "바나나", price: 1500 },
    { name: "오렌지", price: 2000 }
];

const listElement = document.getElementById("item-list");

// 배열 순회하며 HTML 생성
items.forEach(item => {
    const li = document.createElement("li");
    li.textContent = `${item.name}: ${item.price}원`;
    listElement.appendChild(li);
});

        }

</head>
<body>
    <h1>데이터 바인딩 실습</h1>

    <div class="card">
        <h2 id="card-title">로딩 중...</h2>
        <div class="info-row">
            <span class="label">이름:</span>
            <span class="value" id="name">-</span>
        </div>
        <div class="info-row">
            <span class="label">나이:</span>
            <span class="value" id="age">-</span>
        </div>
        <div class="info-row">
            <span class="label">직업:</span>
            <span class="value" id="job">-</span>
        </div>
    </div>

    <script>
        // 데이터 정의
        const userData = {
            title: "내 프로필",
            name: "홍길동",
            age: 25,
            job: "개발자"
        };

        // 데이터 바인딩 함수
        function bindData(data) {
            document.getElementById("card-title").textContent = data.title;
            document.getElementById("name").textContent = data.name;
            document.getElementById("age").textContent = data.age + "세";
            document.getElementById("job").textContent = data.job;
        }

        // 페이지 로드 시 실행
        window.onload = function() {
            bindData(userData);
        };
    </script>
</body>
</html>

        const weatherIcons = {
            "맑음": "☀️",
            "구름 조금": "🌤️",
            "구름 많음": "⛅",

                    return icon;
                }

            return "🌡️";  // 기본 아이콘

        function bindWeatherData(data) {
            // 수집 시간
            document.getElementById("timestamp").textContent =
                `데이터 수집 시간: ${data.collected_at}`;

            // 도시명
            document.getElementById("city-name").textContent =
                `${data.city}, ${data.country || ''}`;

            // 온도
            const temp = data.weather?.temperature || data.temperature;
            document.getElementById("temp").textContent =
                Math.round(temp * 10) / 10;

            // 날씨 설명
            const description = data.weather?.description || data.description;
            document.getElementById("description").textContent = description;

            // 아이콘
            document.getElementById("weather-icon").textContent =
                getWeatherIcon(description);

            // 체감 온도
            const feelsLike = data.weather?.feels_like || data.feels_like;
            document.getElementById("feels-like").textContent =
                Math.round(feelsLike * 10) / 10;

            // 습도
            const humidity = data.weather?.humidity || data.humidity;
            document.getElementById("humidity").textContent = humidity;

            // 풍속
            const wind = data.wind?.speed || data.wind_speed || 0;
            document.getElementById("wind").textContent =
                Math.round(wind * 10) / 10;
        }

        // JSON 파일 로드
        async function loadWeatherData() {

                        "humidity": 62,
                        "description": "맑음"
                    },

                    }
                };

                    '<div class="error">데이터를 불러올 수 없습니다.</div>';
            }
        }

        // 페이지 로드 시 실행
